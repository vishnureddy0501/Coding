	export const getProducts = async (req, res) => {
	try {
		const cacheKey = "products";

		// 1️⃣ Check Redis cache first
		const cached = await redis.get(cacheKey);
		if (cached) {
		console.log("📦 Serving from cache");
		return res.json(JSON.parse(cached));
		}

		// 2️⃣ Simulate DB fetch (slow)
		console.log("🐢 Fetching from DB...");
		const data = productsDB;

		// 3️⃣ Store in Redis (expires in 60s)
		await redis.set(cacheKey, JSON.stringify(data), "EX", 60);

		res.json(data);
	} catch (err) {
		console.error(err);
		res.status(500).json({ error: "Server error" });
	}
	};

	import redis from "../services/redisClient.js";

	// Dummy DB (in-memory for demo)
	let productsDB = [
	{ id: 1, name: "Laptop", price: 800 },
	{ id: 2, name: "Phone", price: 500 },

	// 🔹 Fetch all products (cached as list)

		res.json(data);
	} catch (err) {
		console.error(err);
		res.status(500).json({ error: "Server error" });
	}
	};

	// 🔹 Fetch single product
	export const getProductById = async (req, res) => {
	try {
		const { id } = req.params;
		const cacheKey = `product:${id}`;

		const cached = await redis.get(cacheKey);
		if (cached) {
		console.log(`📦 Serving product ${id} from cache`);
		return res.json(JSON.parse(cached));
		}

		console.log(`🐢 Fetching product ${id} from DB...`);
		const product = productsDB.find((p) => p.id === Number(id));

		if (!product) {
		return res.status(404).json({ error: "Product not found" });
		}

		await redis.set(cacheKey, JSON.stringify(product), "EX", 60);
		res.json(product);
	} catch (err) {
		console.error(err);
		res.status(500).json({ error: "Server error" });
	}
	};

	// 🔹 Update a product (and invalidate cache)
	export const updateProduct = async (req, res) => {
	try {
		const { id } = req.params;
		const { name, price } = req.body;

		let product = productsDB.find((p) => p.id === Number(id));
		if (!product) return res.status(404).json({ error: "Product not found" });

		// Update in "DB"
		product.name = name || product.name;
		product.price = price || product.price;

		// Invalidate caches
		await redis.del(`product:${id}`);
		await redis.del("products");

		console.log(`♻ Cache cleared for product:${id} & products list`);
		res.json(product);
	} catch (err) {
		console.error(err);
		res.status(500).json({ error: "Server error" });
	}
	};

	// 🔹 Delete a product (and invalidate cache)
	export const deleteProduct = async (req, res) => {
	try {
		const { id } = req.params;

		const index = productsDB.findIndex((p) => p.id === Number(id));
		if (index === -1) return res.status(404).json({ error: "Product not found" });

		// Remove from "DB"
		const deleted = productsDB.splice(index, 1);

		// Invalidate caches
		await redis.del(`product:${id}`);
		await redis.del("products");

		console.log(`🗑 Product ${id} deleted. Cache cleared.`);
		res.json({ message: "Product deleted", deleted });
	} catch (err) {
		console.error(err);
		res.status(500).json({ error: "Server error" });
	}
	};

🔧 Helper: Cache Key Builder
// utils/cacheKey.js
export function buildCacheKey(namespace, params) {
  // Sort keys alphabetically to ensure consistency
  const sortedKeys = Object.keys(params).sort();

  const keyParts = sortedKeys.map((k) => `${k}=${params[k]}`);
  return `${namespace}:` + keyParts.join(":");
}


✅ Example:

buildCacheKey("report", { node: "us-east", segment_id: 123, period: "2025-09" });
// "report:node=us-east:period=2025-09:segment_id=123"


Even if query params are passed in different orders, the key will always be the same.

🔧 Controller Example (reportController.js)
import redis from "../services/redisClient.js";
import { buildCacheKey } from "../utils/cacheKey.js";

// Simulated DB/data fetch
const fetchFromDB = async (params) => {
  return {
    ...params,
    revenue: Math.floor(Math.random() * 10000), // demo value
  };
};

export const getReport = async (req, res) => {
  try {
    const params = req.query; // node, period, segment_id, etc.

    // Build unique key dynamically
    const cacheKey = buildCacheKey("report", params);

    // 1️⃣ Check cache
    const cached = await redis.get(cacheKey);
    if (cached) {
      console.log("📦 Serving from cache:", cacheKey);
      return res.json(JSON.parse(cached));
    }

    // 2️⃣ Fetch from DB/external service
    console.log("🐢 Fetching from DB...");
    const data = await fetchFromDB(params);

    // 3️⃣ Save to Redis (TTL = 5 min)
    await redis.set(cacheKey, JSON.stringify(data), "EX", 300);

    res.json(data);
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Server error" });
  }
};
